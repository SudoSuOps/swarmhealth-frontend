Run It Yourself

Every model is available on HuggingFace. Run locally, no cloud required.

# Install
pip install transformers

# Load the Diabetes Companion
from transformers import pipeline

companion = pipeline(
  "text-generation",
  model="swarmhealth/diabetes-companion"
)

# Chat
response = companion("I'm struggling with morning highs...")
print(response)
